
import cv2
import numpy as np
import face_recognition
import psycopg2
from scipy.spatial.distance import euclidean
from ultralytics import YOLO  # Load YOLOv8

# Load YOLOv8 model for face detection
yolo_model = YOLO("/home/indah-mbah/Documents/EduTrack/backend/runs/detect/train/weights/best.pt")

# Connect to PostgreSQL
conn = psycopg2.connect("dbname=edu_track user=postgres password=postgress123 host=localhost")
cursor = conn.cursor()

# Open webcam
cap = cv2.VideoCapture(0)

while True:
    ret, frame = cap.read()
    if not ret:
        break

    # 🔹 Step 1: Use YOLOv8 to Detect Faces
    results = yolo_model(frame)
    boxes = results[0].boxes.xyxy.cpu().numpy()

    for box in boxes:
        x1, y1, x2, y2 = map(int, box)
        face_crop = frame[y1:y2, x1:x2]  # Crop detected face

        # Convert to RGB for face_recognition
        rgb_face = cv2.cvtColor(face_crop, cv2.COLOR_BGR2RGB)

        # 🔹 Step 2: Extract Embedding using face_recognition
        face_locations = face_recognition.face_locations(rgb_face)
        face_encodings = face_recognition.face_encodings(rgb_face, face_locations)

        if face_encodings:
            detected_embedding = face_encodings[0]  # 128-dimensional embedding

            # 🔹 Step 3: Compare with Stored Embeddings in PostgreSQL
            cursor.execute("""
                SELECT s.matricule, s.name, s.level, f.face_embedding 
                FROM students s
                JOIN face_records f ON s.matricule = f.matricule
            """)
            records = cursor.fetchall()

            best_match = None
            best_distance = float("inf")

            for matricule, name, level, stored_embedding in records:
                stored_encoding = np.frombuffer(stored_embedding, dtype=np.float64)[:128]  # Match face_recognition format

                distance = euclidean(stored_encoding, detected_embedding)

                if distance < best_distance:  # Keep track of best match
                    best_distance = distance
                    best_match = (matricule, name, level, distance)

            # 🔹 Step 4: Print Detected Student Info in Terminal
            if best_match and best_distance < 0.4:  # More strict matching
                matricule, name, level, confidence = best_match
                confidence_score = 1 - best_distance  # Convert distance to probability-like score

                print("\n🔹 Detected Student Info:")
                print(f"   Matricule: {matricule}")
                print(f"   Name: {name}")
                print(f"   Level: {level}")
                print(f"   Confidence Score: {confidence_score:.2f}\n")

                cursor.execute("UPDATE students SET last_seen = NOW() WHERE matricule = %s", (matricule,))
                conn.commit()

            # Draw bounding box and overlay student info
            confidence_score = 1 - best_distance  # Define confidence before using it
            text = f"{name} ({matricule}) - Level: {level} - Confidence: {confidence_score:.2f}"

            cv2.putText(frame, text, (x1, y1 - 10), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (0, 255, 0), 2)
            cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)

    cv2.imshow("Live Face Recognition", frame)
    if cv2.waitKey(1) & 0xFF == ord('q'):
        break

cap.release()
cv2.destroyAllWindows()
conn.close()
print("✅ Real-time recognition complete!")
